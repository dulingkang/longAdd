Longadd 大数相加

Two long number add together and output

Download this project as a .zip file Download this project as a tar.gz file

两个不超过128位的数相加，输出和，输入时 中间以空格格开，例如，输入： 1221 4322 输出 5543

using namespace std;

#define MAX_INPUT 129

int charToInt(char source[], int dest[]);
int addLong(int s1[],int s2[],int len1, int len2, int out[]);
void prin(int a[], int len);
void reverserMy(int sourceRe[], int lenRever);


int _tmain(int argc, _TCHAR* argv[])
{
    char a[MAX_INPUT] ;
    char b[MAX_INPUT] ;
    int a1[MAX_INPUT] = {0};
    int b1[MAX_INPUT] = {0};
    int c[MAX_INPUT];

    cin>>a;
    cin>>b;
    int len1 = charToInt(a,a1);
    int len2 = charToInt(b,b1);
    reverserMy(a1,len1);
    reverserMy(b1,len2);
    int len3 = addLong(a1,b1,len1,len2,c);
    reverserMy(c, len3);
    prin(c,len3);
    return 0;
}
int charToInt(char source[], int dest[])
{
    int i = 0;
    char *p = source;
    int destTmp[MAX_INPUT]={0};
    while(*p != '\0')
    {
        if(*p >= '0' && *p <= '9')
        {
            destTmp[i] = *p - '0';
            i++;
            p++;
        }
        else p++;
    }
    for(int j = 0;j < i; j++)
    {
        dest[j] = destTmp[j];
    }
    return i;
}
int addLong(int s1[],int s2[],int len1, int len2, int out[])
{
    int len = len1 < len2 ? len2 : len1;
    int outTmp[MAX_INPUT] = {0};
    int j = 0;
    int value = 0;
    for(int i = 0; i < len; i++)
    {
        outTmp[j++] = (value + s1[i] + s2[i])%10;
        value = (value + s1[i] + s2[i])/10;
    }
    if(value) outTmp[j++] = value;
    for(int m = 0;m < j; m++)
    {
        out[m] = outTmp[m];
    }
    return j;
}

void prin(int a[],int len)
{
    for (int m = 0;m < len; m++)
        cout<<a[m];
}

void reverserMy(int sourceRe[], int lenRever)
{
    int tmp = 0;
    for (int i = 0; i <= (lenRever-1)/2; i++)
    {
        tmp = sourceRe[i];
        sourceRe[i] = sourceRe[lenRever-1-i];
        sourceRe[lenRever-1-i] = tmp;
    }

}
